git commit — добавляет файлы в репозиторий.

git log — просмотр журнала коммитов. 

git show — просмотр коммита. 

git diff — просмотр изменений до коммита. 

git difftool — запуск внешнего инструмента сравнения файлов. 

git restore — отмена изменений. 

git rm — удаление файлов из индекса. 

git reset — откат коммита. 

git branch — создание новой ветки или просмотр веток. 

git merge — слияние репозиториев. 

git remote — привязка локального и удалённого репозитория, просмотр удалённых репозиториев. 

git push — отправка изменений в удалённый репозиторий. 

git pull — получение изменений из удалённого репозитория.

--version - версия git

-h | --help — выводит краткое описание и список наиболее часто используемых команд. Если указать опцию --all или -a, то будут напечатаны все доступные команды.


-C <path> — запускает команду так, как будто Git был запущен в указанном пути, а не в текущем рабочем каталоге.


-c <name>=<value> — передаёт команде параметр конфигурации, который заменяет значения из файлов конфигурации.


--git-dir=<path> — указывает путь к каталогу, в котором будет работать Git.


--work-tree=<path> — указывает путь к рабочему дереву.


--namespace=<name> — устанавливает пространство имён.


--config-env=<name>=<envvar> — устанавливает конфигурацию.
